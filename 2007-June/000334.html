===================================================================
--- src/gpodder/console.py	(revision 332)
+++ src/gpodder/console.py	(working copy)
@@ -101,7 +101,7 @@

-    urlcallback = lambda url: msg( 'update', url)
+    urlcallback = lambda url: msg( 'update', url.replace('%','%%'))
     errorcallback = lambda s: msg( 'error', s)
 
     return gPodderChannelReader().read( True, callback_url =
urlcallback, callback_error = errorcallback)

will be available in the next release.
